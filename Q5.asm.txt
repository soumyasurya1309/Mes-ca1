ORG 0000H

MOV R0,#40H
MOV R1,#40H
MOV R2,#20H

SCAN:
MOV A,@R0
CJNE A,#0FFH,VALID
SJMP SKIP

VALID:
MOV @R1,A
INC R1

SKIP:
INC R0
DJNZ R2,SCAN

MOV A,R1
CLR C
SUBB A,#60H
JZ STOP

MOV R2,A
CPL A
INC A
ADD A,#20H
MOV R2,A

FILL:
MOV @R1,#00H
INC R1
DJNZ R2,FILL

STOP:
END
